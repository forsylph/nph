## 3. Rule 실행 패턴

### 3.1 기본 Rule 실행 패턴 (ClamChkRuleUC)

**출처**: `nph/his/hp/com/uc/ClamChkRuleUC.java` (보험심사 UC)

```java
package nph.his.hp.com.uc;

import com.innoexpert.rulesclient.ConnectionProperties;      // [✅ 확인됨]
import com.innoexpert.rulesclient.Constants;                 // [✅ 확인됨]
import com.innoexpert.rulesclient.ItemValue;               // [✅ 확인됨]
import com.innoexpert.rulesclient.ResultSet;               // [✅ 확인됨]
import com.innoexpert.rulesclient.ResultSetMetaData;       // [✅ 확인됨]
import com.innoexpert.rulesclient.RuleReq;                 // [✅ 확인됨]
import com.innoexpert.rulesclient.RulesException;          // [✅ 확인됨]
import com.innoexpert.rulesclient.RulesInterface;          // [✅ 확인됨]

import devon.core.collection.LData;
import devon.core.collection.LMultiData;

public class ClamChkRuleUC {

    /**
     * 보험심사 Rule 실행
     * [✅ 확인됨] ClamChkRuleUC.java
     */
    public LMultiData chkClamRule(LData data) throws LException {
        LMultiData rMData = new LMultiData();
        String sRuleCode = "";

        RulesInterface intf = null;

        try {
            // 1. Rule 요청 객체 생성
            RuleReq req = new RuleReq();
            ResultSet rs = null;
            ItemValue item;
            ResultSetMetaData rsmd;
            int iColCnt;
            int iRuleCodeType = Constants.CODETYPE_ALIAS;  // [✅ 확인됨]

            // 2. Rule 이름 설정 (동적)
            sRuleCode = data.getString("ruleCode");        // [✅ 확인됨] 예: "#S00000327"
            req.setRuleCode(sRuleCode);                      // [✅ 확인됨]

            // 3. 실행 기준일 설정
            req.setDate("2014-01-20");                       // [✅ 확인됨]

            // 4. 파라미터 설정
            req.resetItems();

            // 등록번호
            item = new ItemValue();                          // [✅ 확인됨]
            item.setCode("PID");                             // [✅ 확인됨]
            item.addStrValue(data.getString("pid"));         // [✅ 확인됨]
            req.addItem(item);                               // [✅ 확인됨]

            // 내원번호
            item = new ItemValue();
            item.setCode("CHOS_NO");
            item.addStrValue(data.getString("chosNo"));
            req.addItem(item);

            // 용어코드
            item = new ItemValue();
            item.setCode("TMLG_CDID");
            item.addStrValue(data.getString("tmlgCdId"));
            req.addItem(item);

            // 진료구분
            item = new ItemValue();
            item.setCode("MED_DVSN");
            item.addStrValue(data.getString("medDvsn"));
            req.addItem(item);

            // 환자유형
            item = new ItemValue();
            item.setCode("INS_TYPE_CD");
            item.addStrValue(data.getString("insTypeCd"));
            req.addItem(item);

            // 연령
            item = new ItemValue();
            item.setCode("AGE");
            item.addNumValue(data.getInt("age"));
            req.addItem(item);

            // 성별
            item = new ItemValue();
            item.setCode("SEX");
            item.addStrValue(data.getString("sex"));
            req.addItem(item);

            // 수가코드
            item = new ItemValue();
            item.setCode("SUGA_CD");
            item.addStrValue(data.getString("snglCd"));
            req.addItem(item);

            // 표준코드
            item = new ItemValue();
            item.setCode("EDI_CD");
            item.addStrValue(data.getString("ediCd"));
            req.addItem(item);

            // ... 기타 파라미터 설정

            // 5. Rule Execution
            intf = ConnectionProperties.getConnection();     // [✅ 확인됨]
            rs = intf.execute(req, iRuleCodeType);          // [✅ 확인됨]

            // 6. 결과 처리
            rsmd = rs.getMetaData();                         // [✅ 확인됨]
            iColCnt = rsmd.getColumnCount();

            while (rs.next()) {                              // [✅ 확인됨]
                for (int i = 1; i <= iColCnt; i++) {
                    if (i == 1) {
                        rMData.add("R_JOB_GUBN", rs.getString(i));
                    } else if (i == 2) {
                        rMData.add("R_MAIN_SUB", rs.getString(i));
                    } else if (i == 3) {
                        rMData.add("R_CODE", rs.getString(i));
                    } else if (i == 4) {
                        rMData.add("R_MSG", rs.getString(i));
                    } else if (i == 5) {
                        rMData.add("R_SNGL_CD", rs.getString(i));
                    }
                }
            }

            return rMData;

        } catch (RulesException re) {                        // [✅ 확인됨]
            LLog.err.write(re.getMessage());
            if(intf != null)
                intf.close();                                 // [✅ 확인됨]
            throw new BizException(re.getErrorCode() + "\n" + re.getMessage());
        } catch (IndexOutOfBoundsException e) {
            LLog.err.write(e.getMessage());
            if(intf != null)
                intf.close();
            throw new LException(e);
        } finally {
            if(intf != null)
                intf.close();
            return rMData;
        }
    }
}
```

### 3.5 Batch 연동 패턴

```java
// [⚠️ 추정] 배치 모듈에서의 사용 예시
package nph.bat.sp.reader;

import com.innoexpert.innorulesj.client.IRClient;
import com.innoexpert.innorulesj.session.IRSession;
import devonframework.batch.item.reader.AbstractItemReader;

public class RuleBasedItemReader extends AbstractItemReader {

    private IRClient ruleClient;
    private IRSession ruleSession;

    @Override
    public void initialize() {
        // 배치 초기화 시 Rule Engine 연결
        ruleClient = IRClient.getInstance();
        ruleSession = ruleClient.createSession("rule");
    }

    @Override
    public Object read() {
        // 데이터 읽기
        LData inputData = readFromSource();

        // Rule 적용
        ruleSession.setInputData(convertToIRData(inputData));
        IRData result = ruleSession.execute("BATCH_PROCESSING_RULE");

        return convertToOutput(result);
    }

    @Override
    public void close() {
        // 세션 종료
        if (ruleSession != null) {
            ruleSession.close();
        }
    }
}
```

```sql

SELECT rule_code, rule_name, rule_content,
       input_params, output_params, version, status
```
